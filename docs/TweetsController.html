def tweet
  tweet = Tweet.new(comment: tweet_params[:comment], user_id: @current_user.id)

  unless tweet.nil?
    authorize tweet
    tweet.comment = extract_link(tweet.comment)
  end

  if tweet.save
    tclient.update( tweet.comment )
    render json: tweet
  else
    render json: {error: true, message: tweet.errors}
  end
end
